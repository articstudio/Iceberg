217:                 $url = empty($url) && !empty($link['url']) ? current($link['url']) : $url;
218:             }
219:             ?>
220:             <li>

222:                 <?php if ($link['submenu']==1 && $page!=null): ?>
223:                 <ul>

225:                     <?php foreach ($childs AS $kc=>$child): ?>
