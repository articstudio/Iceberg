283:     /**
284:      * Environment loader
285:      */
286:     public function LoadEnvironment()
287:     {
288:         $class = null;
289:         foreach (self::$ENVIRONMENT_LIST AS $env)
290:         {
291:             if (function_exists($env['function']))
292:             {
293:                 $in = $env['function']();
294:                 if ($in)
295:                 {
296:                     $class = $env['class'];
297:                     break;
298:                 }
299:             }
300:         }
301:         if (!is_null($class) && class_exists($class)) {
302:             $this->Environment = new $class();
303:             $this->Environment
304:                     ->Load()
305:                     ->Config()
306:                     ->Generate()
307:                     ->Show();
308:             $this->time_end = microtime(true);
309:             $time = $this->time_end - $this->time_start;
310:             if (ICEBERG_DEBUG_MODE)
311:             {
312:                 print "\n\n" . '<!-- Page generated in ' . $time . ' seconds -->';
313:                 MySQL::PrintLog();
314:                 IcebergCache::PrintLog();
315:             }
316:         }
317:         else {
318:             throw new IcebergException('ICEBERG INITIALIZATION ERROR: Environment not found.');
319:         }
320:     }
