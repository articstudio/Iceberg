 41:         global $__EXTENSIONS;
 42:         $dir = CMS_DIR_EXTENSIONS;
 43:         $dh = opendir($dir);

 50:                         $info = array(
 51:                             'name' => '',
 52:                             'description' => '',
 53:                             'url' => '',
 54:                             'author' => '',

 56:                         );

 58:                         if (array_key_exists($file, $__EXTENSIONS)) {
 59:                             $active = $__EXTENSIONS[$file]['active'];
 60:                             $__EXTENSIONS[$file] = array_merge($__EXTENSIONS[$file], $info);
 61:                             $__EXTENSIONS[$file]['active'] = $active;
 62:                         }
 63:                         else {
 64:                             $__EXTENSIONS[$file] = $info;
 65:                             $__EXTENSIONS[$file]['active'] = false;
 66:                         }
 67:                     }
 68:                 }
 69:             }
